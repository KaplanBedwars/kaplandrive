// Middleware
app.use(cors());
app.use(express.json());
app.use('/uploads', express.static(path.join(__dirname, 'uploads')));

// Dosya y√ºkleme ayarlarƒ±
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    cb(null, 'uploads/');
  },
  filename: (req, file, cb) => {
    cb(null, file.originalname);
  },
});
const upload = multer({ storage });

// /files ‚Üí dosya listesini JSON olarak g√∂nderir
app.get('/files', (req, res) => {
  fs.readdir('uploads', (err, files) => {
    if (err) return res.status(500).send('Dosyalar okunamadƒ±');
    res.json({ files: files.map(filename => `/uploads/${filename}`) });
  });
});

// /upload ‚Üí dosya y√ºkler
app.post('/upload', upload.single('file'), (req, res) => {
  if (!req.file) {
    return res.status(400).send({ error: 'Dosya se√ßilmedi.' });
  }
  res.status(200).json({ url: `/uploads/${req.file.filename}` });
});

// /rename ‚Üí dosya yeniden adlandƒ±rƒ±r
app.post('/rename', (req, res) => {
  const { oldPath, newName } = req.body;

  const sanitizedOldPath = oldPath.replace(/^\/?uploads\//, '');
  if (sanitizedOldPath.includes('..') || newName.includes('..')) {
    return res.status(400).json({ error: "Ge√ßersiz dosya yolu!" });
  }

  const baseDir = path.join(__dirname, 'uploads');
  const absoluteOldPath = path.join(baseDir, sanitizedOldPath);
  const absoluteNewPath = path.join(baseDir, path.dirname(sanitizedOldPath), newName);

  if (!fs.existsSync(absoluteOldPath)) {
    return res.status(404).json({ error: "Dosya bulunamadƒ±!" });
  }

  fs.rename(absoluteOldPath, absoluteNewPath, (err) => {
    if (err) {
      console.error("Hata:", err);
      return res.status(500).json({ error: "Dosya i≈ülenemedi!" });
    }
    res.json({ message: "Dosya adƒ± deƒüi≈ütirildi", newName });
  });
});

app.listen(PORT, () => {
  console.log(`üì° KaplanDrive Sunucusu http://localhost:${PORT} adresinde √ßalƒ±≈üƒ±yor`);
